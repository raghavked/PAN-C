### Task 3.1: Add Push Token Storage Endpoint

**Reference**: Ask user to show `server/src/routes/user.ts` or equivalent

**Action**: Add endpoint to store Expo push tokens

```typescript
// File: server/src/routes/user.ts
// Add this endpoint to the router:

// POST /api/user/push-token
// Register device's Expo push token
router.post('/push-token', async (req: any, res: any) => {
  try {
    const userEmail = req.user?.email;
    if (!userEmail) {
      return res.status(401).json({ error: 'Not authenticated' });
    }

    const { pushToken } = req.body;

    if (!pushToken) {
      return res.status(400).json({ error: 'Push token required' });
    }

    // Validate Expo token format
    if (!pushToken.startsWith('ExponentPushToken[')) {
      return res.status(400).json({ error: 'Invalid push token format' });
    }

    // Store in MongoDB
    const result = await db.collection('users').updateOne(
      { email: userEmail },
      {
        $set: {
          pushToken: pushToken,
          pushTokenUpdatedAt: new Date(),
          pushTokenType: 'ExponentPushToken',
        },
      }
    );

    if (result.matchedCount === 0) {
      return res.status(404).json({ error: 'User not found' });
    }

    console.log(`✅ Push token registered for ${userEmail}`);

    res.json({
      success: true,
      message: 'Push token registered',
      tokenRegistered: pushToken.substring(0, 50) + '...',
    });
  } catch (error: any) {
    console.error('Push token error:', error);
    res.status(500).json({ success: false, error: error.message });
  }
});
```

### Task 3.2: Update Panic Endpoint for Expo Push

**Reference**: Ask user to show `server/src/routes/panic.ts`

**Action**: Install Expo SDK and update panic trigger

```bash
cd server

# Install expo-server-sdk
```

Then update panic endpoint:

```typescript
// File: server/src/routes/panic.ts
import * as Expo from 'expo-server-sdk';

// Initialize Expo
const expo = new Expo.Expo({
  accessToken: process.env.EXPO_ACCESS_TOKEN,
});

// Update the triggerPanic endpoint:
router.post('/panic/trigger', async (req: any, res: any) => {
  try {
    const userEmail = req.user?.email;
    if (!userEmail) {
      return res.status(401).json({ error: 'Not authenticated' });
    }

    const { latitude, longitude, address } = req.body;

    if (!latitude || !longitude || !address) {
      return res.status(400).json({
        error: 'Missing location data',
        required: ['latitude', 'longitude', 'address'],
      });
    }

    // Get user and contacts
    const user = await db.collection('users').findOne({ email: userEmail });
    if (!user) return res.status(404).json({ error: 'User not found' });

    const contacts = await db.collection('contacts')
      .find({ userEmail: userEmail })
      .toArray();

    console.log(`[PANIC] ${userEmail} triggered panic`);
    console.log(`[PANIC] Found ${contacts.length} contacts`);

    // Get push tokens
    const pushTokens = contacts
      .filter((c: any) => c.pushToken && c.pushToken.startsWith('ExponentPushToken['))
      .map((c: any) => c.pushToken);

    console.log(`[PANIC] ${pushTokens.length} contacts have Expo push tokens`);

    // Create incident record
    const incident = {
      userEmail: userEmail,
      userName: user.fullName,
      userPhone: user.phone,
      triggeredAt: new Date(),
      location: {
        latitude,
        longitude,
        address,
      },
      contactsNotified: contacts.map((c: any) => ({
        contactId: c._id,
        contactName: c.name,
        pushToken: c.pushToken || null,
        notifiedAt: new Date(),
        status: c.pushToken ? 'pending' : 'no_app',
      })),
      status: 'active',
      sentCount: 0,
      failedCount: 0,
    };

    const incidentResult = await db.collection('incidents').insertOne(incident);
    const incidentId = incidentResult.insertedId.toString();

    // Send Expo Push Notifications
    let sentCount = 0;
    let failedCount = 0;
    const messages: Expo.ExpoPushMessage[] = [];

    for (const token of pushTokens) {
      if (!Expo.Expo.isExpoPushToken(token)) {
        console.error(`[PANIC] Invalid token: ${token}`);
        failedCount++;
        continue;
      }

      messages.push({
        to: token,
        sound: 'default',
        title: `🚨 ${user.fullName} NEEDS HELP!`,
        body: `Location: ${address}`,
        data: {
          incidentId: incidentId,
          userEmail: userEmail,
          userName: user.fullName,
          address: address,
          latitude: latitude.toString(),
          longitude: longitude.toString(),
        },
        priority: 'high',
        badge: 1,
      });
    }

    // Send via Expo
    if (messages.length > 0) {
      const chunks = expo.chunkPushNotifications(messages);
      for (const chunk of chunks) {
        try {
          const tickets = await expo.sendPushNotificationsAsync(chunk);
          for (const ticket of tickets) {
            if (ticket.status === 'ok') {
              sentCount++;
              console.log(`[EXPO] ✅ Message sent: ${ticket.id}`);
            } else {
              failedCount++;
              console.log(`[EXPO] ❌ Failed: ${ticket.message}`);
            }
          }
        } catch (error) {
          console.error('[EXPO] Error:', error);
          failedCount += chunk.length;
        }
      }
    }

    // Update incident
    await db.collection('incidents').updateOne(
      { _id: new ObjectId(incidentId) },
      {
        $set: {
          status: sentCount > 0 ? 'notified' : 'pending',
          sentAt: new Date(),
          sentCount: sentCount,
          failedCount: failedCount,
        },
      }
    );

    console.log(`[PANIC] ✅ Sent ${sentCount}, Failed ${failedCount}`);

    res.json({
      success: true,
      incidentId: incidentId,
      contactsNotified: sentCount,
      contactsFailed: failedCount,
      totalContacts: contacts.length,
      status: 'active',
      message: `Alert sent to ${sentCount} contact${sentCount !== 1 ? 's' : ''}`,
    });
  } catch (error: any) {
    console.error('[PANIC] Error:', error);
    res.status(500).json({ success: false, error: error.message });
  }
});
```

## PHASE 4: REACT NATIVE FRONTEND (10 minutes)

### Task 4.1: Create Push Notifications Service

**Action**: Create new file `rn/src/services/pushNotifications.ts`

```typescript
import * as Notifications from 'expo-notifications';
import * as Device from 'expo-device';

// Configure notification handler
Notifications.setNotificationHandler({
  handleNotification: async () => ({
    shouldShowAlert: true,
    shouldPlaySound: true,
    shouldSetBadge: true,
  }),
});

export async function registerForPushNotificationsAsync(): Promise<string | null> {
  if (!Device.isDevice) {
    console.log('Must use physical device for push notifications');
    return null;
  }

  try {
    // Request permissions
    const { status: existingStatus } = await Notifications.getPermissionsAsync();
    let finalStatus = existingStatus;

    if (existingStatus !== 'granted') {
      const { status } = await Notifications.requestPermissionsAsync();
      finalStatus = status;
    }

    if (finalStatus !== 'granted') {
      console.log('Push permissions not granted');
      return null;
    }

    // Get Expo push token
    const projectId = process.env.EXPO_PROJECT_ID;
    if (!projectId) {
      throw new Error('EXPO_PROJECT_ID not set in environment');
    }

    const token = await Notifications.getExpoPushTokenAsync({ projectId });
    console.log('✅ Expo Push Token:', token.data);
    return token.data;
  } catch (error) {
    console.error('Failed to get push token:', error);
    return null;
  }
}

export function setupNotificationListeners() {
  const subscription = Notifications.addNotificationResponseReceivedListener((response) => {
    console.log('🔔 Notification tapped:', response.notification);
    const { incidentId } = response.notification.request.content.data as any;
    if (incidentId) {
      // Navigate to incident details if needed
      console.log('Opening incident:', incidentId);
    }
  });

  return subscription;
}
```

```typescript
import { registerForPushNotificationsAsync } from '../services/pushNotifications';

useEffect(() => {
  // Register push token whenever auth changes
  if (user) {
    registerPushTokenWithBackend();

// Add this function:
const registerPushTokenWithBackend = async () => {
  try {
    const token = await registerForPushNotificationsAsync();
    
    if (token) {
      console.log('✅ Push token obtained');
      
      // Send to backend
      const authToken = await AsyncStorage.getItem('authToken');
      const response = await fetch('/api/user/push-token', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${authToken}`,
        },
        body: JSON.stringify({ pushToken: token }),
      });

      if (response.ok) {
        console.log('✅ Push token registered with backend');
      } else {
        console.error('Failed to register push token');
      }
    }
  } catch (error) {
    console.error('Push token registration error:', error);
    // Don't block auth if push registration fails
  }
};
```

## TROUBLESHOOTING GUIDE FOR AGENT

If tests fail, investigate in this order:

1. **"No push token obtained"**
   - Check: `EXPO_PROJECT_ID` in Replit Secrets
   - Check: `rn/.env` has EXPO_PROJECT_ID
   - Check: Device has notifications enabled
   - Check: Using physical device (not simulator)

2. **"Push token not in MongoDB"**
   - Check: Backend is running (`npm run dev`)
   - Check: `MONGODB_URI` is correct
   - Check: Network connection is working
   - Check: `/api/user/push-token` endpoint responds

3. **"Panic doesn't trigger at zero"**
   - Check: `TimerScreen.tsx` has useEffect watching `localSeconds`
   - Check: `triggerPanic()` function exists
   - Check: No console errors

4. **"No notification on other device"**
   - Check: Receiver has valid push token
   - Check: `EXPO_ACCESS_TOKEN` is set (if needed)
   - Check: Both devices on same network or accessible
   - Check: Notification permissions enabled

5. **"Compilation errors"**
   - Run: `npm run build` in server directory
   - Run: `npx tsc --noEmit` in rn directory
   - Check: All imports are correct
   - Check: No typos in endpoint names
